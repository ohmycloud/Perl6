# 解构

my $data = ("one", "two", "three");
my ($one, $two, $three) = $data;
say $one;

# 返回多个值的函数
sub getPerson() {
    return ("Taylor Swift", 26);
}

my ($name, $age) = getPerson();
say $name;

# 在解构时忽略某个值
my ($name, $) = getPerson();
say $name;
